    };

      // cleanup
      watcher.offLayout(handler);
    };
  }, []);

    // You could also use LayoutWatcher directly and show/hide the element in
    // your callback
    const msg = msgRef.current;
    let trigger: LayoutTrigger;
    if (msg) {

        layout: "max tablet",

    return () => {
      // cleanup
